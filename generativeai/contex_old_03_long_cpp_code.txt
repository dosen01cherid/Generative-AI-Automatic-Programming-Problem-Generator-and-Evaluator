Example 27
------------
Program implementing stack using linked list
------------
#include <iostream>
using namespace std;
struct Node{
   int data;
   Node *next;
};
class Stack{
   Node *top;
public:
   Stack(){
      top = NULL;
   }
   void push(int value){
      Node *newNode = new Node;
      newNode->data = value;
      newNode->next = top;
      top = newNode;
   }
   int pop(){
      if(top == NULL) return -1;
      int value = top->data;
      Node *temp = top;
      top = top->next;
      delete temp;
      return value;
   }
};
int main(){
   Stack s;
   s.push(5);
   s.push(10);
   cout << "Popped: " << s.pop() << endl;
   return 0;
}

Program implementing queue using array
------------
#include <iostream>
using namespace std;
class Queue{
   int arr[100];
   int front, rear;
public:
   Queue(){
      front = -1;
      rear = -1;
   }
   void enqueue(int x){
      if(front == -1) front = 0;
      arr[++rear] = x;
   }
   int dequeue(){
      int value = arr[front];
      front++;
      return value;
   }
};
int main(){
   Queue q;
   q.enqueue(10);
   q.enqueue(20);
   cout << "Dequeued: " << q.dequeue() << endl;
   return 0;
}

------------
Example 29
------------
Program implementing queue using linked list
------------
#include <iostream>
using namespace std;
struct Node{
   int data;
   Node *next;
};
class Queue{
   Node *front, *rear;
public:
   Queue(){
      front = rear = NULL;
   }
   void enqueue(int value){
      Node *newNode = new Node;
      newNode->data = value;
      newNode->next = NULL;
      if(rear == NULL){
         front = rear = newNode;
         return;
      }
      rear->next = newNode;
      rear = newNode;
   }
   int dequeue(){
      if(front == NULL) return -1;
      int value = front->data;
      Node *temp = front;
      front = front->next;
      if(front == NULL) rear = NULL;
      delete temp;
      return value;
   }
};
int main(){
   Queue q;
   q.enqueue(5);
   q.enqueue(10);
   cout << "Dequeued: " << q.dequeue() << endl;
   return 0;
}

When creating new questions:

FOR FILL-IN-THE-BLANK (MULTIPLE CHOICE FORMAT):
1. Always start with valid C++ code
2. Select ONLY ONE strategic word or character to blank out with _____
3. Provide exactly 4 multiple choice options
4. Focus on testing:
   - Basic keywords: int, if, for, while, return, cout, cin, endl, namespace, include, else, try, catch, throw
   - Type keywords: struct, class, void, float, char, string, bool
   - Memory keywords: new, delete, NULL
   - Access specifiers: public, private, protected
   - Single characters: ; { } " ' >> << -> . * & : ++ -- :: [] ()
5. Include good distractors (related keywords/operators that could confuse students)
6. Format:
   - Code section (complete working code)
   - Question section (code with one _____ blank)
   - Options section (1. option1  2. option2  3. option3  4. option4)
   - Answer section (single number: 1, 2, 3, or 4)
7. Create separate questions for both keyword blanks AND single character blanks

FOR LINE ORDERING:
1. Start with complete, valid C++ code
2. Separate each line (including braces, struct/class definitions)
3. Shuffle randomly but ensure every line is numbered sequentially
4. The answer should recreate syntactically correct, compilable code
5. Format answer as comma-separated numbers: 3,1,2,5,4,6
6. For complex code (classes, structs, functions), ensure proper structure:
   - Headers come first
   - Declarations before definitions
   - Class/struct definition before usage
   - Function definition before main() or after with declaration
7. Pay attention to scope and ordering dependencies

IMPORTANT CONCEPTS TO TEST:
- Memory management: new, delete, pointers
- Data structures: arrays, structs, linked lists, stacks, queues, trees
- OOP: classes, objects, constructors, inheritance, encapsulation
- Access specifiers: public, private, protected
- Pointer operations: *, &, ->, NULL
- ADT operations: push, pop, enqueue, dequeue, insert, delete, traverse
- Tree operations: inorder, preorder, postorder traversal

Remember: Always ensure the original code compiles correctly before creating questions!
