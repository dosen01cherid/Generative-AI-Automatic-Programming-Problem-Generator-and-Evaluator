Example 27
------------
Program implementing stack using linked list
------------
#include <iostream>
using namespace std;
struct Node{
   int data;
   Node *next;
};
class Stack{
   Node *top;
public:
   Stack(){
      top = NULL;
   }
   void push(int value){
      Node *newNode = new Node;
      newNode->data = value;
      newNode->next = top;
      top = newNode;
   }
   int pop(){
      if(top == NULL) return -1;
      int value = top->data;
      Node *temp = top;
      top = top->next;
      delete temp;
      return value;
   }
};
int main(){
   Stack s;
   s.push(5);
   s.push(10);
   cout << "Popped: " << s.pop() << endl;
   return 0;
}

Program implementing queue using array
------------
#include <iostream>
using namespace std;
class Queue{
   int arr[100];
   int front, rear;
public:
   Queue(){
      front = -1;
      rear = -1;
   }
   void enqueue(int x){
      if(front == -1) front = 0;
      arr[++rear] = x;
   }
   int dequeue(){
      int value = arr[front];
      front++;
      return value;
   }
};
int main(){
   Queue q;
   q.enqueue(10);
   q.enqueue(20);
   cout << "Dequeued: " << q.dequeue() << endl;
   return 0;
}

------------
Example 29
------------
Program implementing queue using linked list
------------
#include <iostream>
using namespace std;
struct Node{
   int data;
   Node *next;
};
class Queue{
   Node *front, *rear;
public:
   Queue(){
      front = rear = NULL;
   }
   void enqueue(int value){
      Node *newNode = new Node;
      newNode->data = value;
      newNode->next = NULL;
      if(rear == NULL){
         front = rear = newNode;
         return;
      }
      rear->next = newNode;
      rear = newNode;
   }
   int dequeue(){
      if(front == NULL) return -1;
      int value = front->data;
      Node *temp = front;
      front = front->next;
      if(front == NULL) rear = NULL;
      delete temp;
      return value;
   }
};
int main(){
   Queue q;
   q.enqueue(5);
   q.enqueue(10);
   cout << "Dequeued: " << q.dequeue() << endl;
   return 0;
}

HOW TO CREATE ACCURATE FILL-IN-THE-BLANK QUESTIONS
============================================================

STEP-BY-STEP INSTRUCTIONS:

1. START WITH VALID C++ CODE
   - The code MUST compile and run correctly
   - Keep the code simple and focused
   - Ensure the code demonstrates a clear concept

2. IDENTIFY WHAT TO REPLACE
   - Choose EXACTLY ONE keyword, operator, or symbol to replace
   - Common targets: keywords (int, for, if, class, return, etc.)
   - Operators: =, ==, <<, >>, ++, --, etc.
   - Symbols: ; { } ( ) [ ] < >
   - Method names: push_back, size, length, etc.
   - Header files: iostream, vector, string, etc.

3. CREATE THE BLANK
   - Replace the chosen word/symbol with EXACTLY: _____
   - Use exactly 5 underscores
   - Do NOT change anything else in the code
   - Keep all spacing, indentation, and formatting identical

4. CREATE 4 MULTIPLE CHOICE OPTIONS
   - Option 1: A plausible wrong answer (related keyword/operator)
   - Option 2: THE CORRECT ANSWER (always position 2)
   - Option 3: Another plausible wrong answer
   - Option 4: Another plausible wrong answer

   Wrong answers should be:
   - Related to the concept (similar keywords, common mistakes)
   - Syntactically plausible (could look correct to a beginner)
   - Not obviously wrong (avoid nonsense options)

5. ANSWER FORMAT
   - Always provide the answer as a single number: 2
   - The correct answer is ALWAYS in position 2

6. STRUCTURE FORMAT
   Must include these sections in this exact order:
   ------------------
   Fill-in-the-Blank Question Example [ID] ([Description])
   ------------------
   Code
   ------------------
   [Complete valid C++ code]
   -------------------
   Question
   -------------------
   [Same code with ONE _____ replacement]
   -------------------
   Options
   -------------------
   1. [wrong]  2. [CORRECT]  3. [wrong]  4. [wrong]
   -----------------
   Answer
   -----------------
   2

 int x = 10;          _____ x = 10;           (replace "int")
 cout << "Hello";     _____ << "Hello";        (replace "cout")
 for(int i = 0...)    _____(int i = 0...)     (replace "for")
 if(x > 5)            if(x _____ 5)           (replace ">")
 return 0;            _____ 0;                (replace "return")
 int x = 10;          int x _____ 10;         (replace "=")
 cout << x;           cout _____ x;           (replace "<<")

EXAMPLES OF BAD REPLACEMENTS (AVOID THESE):

1. MISMATCH: Blank and Answer Don't Match
    Code: cout << "Hello" << endl;
    Replace: _____ << "Hello" << endl;
    Wrong Answer: "cout <<"
    Problem: You removed "cout" but answer includes "<<"
    Correct Answer: "cout" (only what was replaced)

    Code: int x = 10;
    Replace: int _____ = 10;
    Wrong Answer: "x ="
    Problem: You removed "x" but answer includes "="
    Correct Answer: "x" (only what was replaced)

    Code: for(int i = 0; i < 5; i++)
    Replace: for(_____ i = 0; i < 5; i++)
    Wrong Answer: "int i"
    Problem: You removed "int" but answer includes "i"
    Correct Answer: "int" (only what was replaced)

    Code: if(x > 5)
    Replace: if(_____ > 5)
    Wrong Answer: "x >"
    Problem: You removed "x" but answer includes ">"
    Correct Answer: "x" (only what was replaced)

    Code: return 0;
    Replace: _____ 0;
    Wrong Answer: "return 0"
    Problem: You removed "return" but answer includes "0"
    Correct Answer: "return" (only what was replaced)

2. REPLACING MULTIPLE WORDS/SYMBOLS AT ONCE
    Code: int x = 10;
    Replace: _____ = 10;
    Problem: Removed both "int" and "x" (two words)
    Correct: Choose only one - either "int" or "x"

    Code: cout << x << endl;
    Replace: cout _____ endl;
    Problem: Removed "<< x <<" (multiple parts)
    Correct: Replace only one operator at a time

3. REPLACING PARTS OF WORDS
    Code: return 0;
    Replace: _____ turn 0;
    Problem: Replacing "re" from "return" (incomplete word)
    Correct: Replace the entire word "return"

    Code: #include <iostream>
    Replace: #_____ clude <iostream>
    Problem: Replacing "in" from "include" (partial word)
    Correct: Replace the entire word "include"

4. REPLACING VARIABLE NAMES OR VALUES
    Code: int sum = 10;
    Replace: int _____ = 10;
    Problem: Replacing variable name "sum" (not educational)
    Correct: Replace keywords, operators, or data types

    Code: int x = 10;
    Replace: int x = _____;
    Problem: Replacing the value "10" (not a concept)
    Correct: Replace "int" or "=" instead

5. MAKING THE BLANK AMBIGUOUS
    Code: int x = 10;

    Problem: Could be "int x" or just "x" (ambiguous)
    Correct: Make it clear what's missing

When creating new questions:

FOR FILL-IN-THE-BLANK (MULTIPLE CHOICE FORMAT):
1. Always start with valid C++ code
2. Select ONLY ONE strategic word or character to blank out with _____
3. Provide exactly 4 multiple choice options
4. Focus on testing:
   - Basic keywords: int, if, for, while, return, cout, cin, endl, namespace, include, else, try, catch, throw
